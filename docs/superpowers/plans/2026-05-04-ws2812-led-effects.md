> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 9 颗 WS2812 灯珠（1 中心 + 8 外圈）实现分层灯效系统，与音效触发同步，支持开关/亮度/速度/风格/动效设置和演示模式。

**Tech Stack:** ESP-IDF v5.5.4, RMT TX driver, esp_timer, NVS, LVGL v9.5.0

```c
void ws2812_set_pixels(const rgb_t *pixels, uint8_t count)
{
    if (!rmt_chan || !encoder || !pixels || count == 0) return;
    // WS2812 expects GRB byte order
    uint8_t *grb = malloc(count * 3);
    if (!grb) return;
    for (int i = 0; i < count; i++) {
        grb[i * 3 + 0] = pixels[i].g;
        grb[i * 3 + 1] = pixels[i].r;
        grb[i * 3 + 2] = pixels[i].b;
    }
    rmt_transmit_config_t tx_cfg = { .loop_count = 0 };
    rmt_transmit(rmt_chan, encoder, grb, count * 3, &tx_cfg);
    rmt_tx_wait_all_done(rmt_chan, -1);
    free(grb);
}
```

```c
#include "led_service.h"
#include "driver/ws2812.h"
#include "service/storage_service.h"
#include "esp_log.h"
#include "esp_timer.h"
#include <math.h>
#include <string.h>

static const char *TAG = "LED";

#define LED_COUNT       9
#define CENTER_IDX      0
#define OUTER_COUNT     8
#define FRAME_INTERVAL_MS 50
#define BRIGHTNESS_LEVELS 10

// Demo colors
const led_color_t led_demo_colors[LED_DEMO_COLOR_COUNT] = {
    {255, 107, 107},  // Red (Work)
    {76, 175, 80},    // Green (Break)
    {77, 150, 255},   // Blue (Long Break)
    {255, 255, 0},    // Yellow (Paused)
    {255, 80, 120},   // Pink (Sad)
};
const char *const led_demo_color_names[LED_DEMO_COLOR_COUNT] = {
    "Red", "Green", "Blue", "Yellow", "Pink"
};

// Settings
static bool enabled = true;
static uint8_t brightness_level = 5;          // 1-10
static led_anim_t animation = LED_ANIM_BREATH;
static led_style_t style = LED_STYLE_PURE;
static led_speed_t speed = LED_SPEED_MEDIUM;

// State machine
typedef enum {
    LED_STATE_IDLE,
    LED_STATE_PLAYING_OUTER,   // Outer ring playing 1 round
    LED_STATE_WAITING_CENTER,  // Center LED looping
    LED_STATE_DEMO,            // Demo mode
} led_state_t;

static led_state_t led_state = LED_STATE_IDLE;
static led_color_t current_color = {0, 0, 0};
static led_color_t demo_color = {0, 0, 0};
static led_color_t saved_wait_color = {0, 0, 0};
static bool had_wait_before_play = false;

// Timer
static esp_timer_handle_t frame_timer = NULL;
static int64_t anim_start_us = 0;
static int frame_count = 0;

// Brightness mapping: level 1-10 → value 26-255
static uint8_t brightness_value(void)
{
    if (brightness_level == 0) return 0;
    return (uint8_t)((brightness_level * 255 + 5) / 10);
}

// Period in microseconds for each speed
static int64_t period_us(void)
{
    switch (speed) {
        case LED_SPEED_SLOW:   return 3000000LL;
        case LED_SPEED_FAST:   return 800000LL;
        default:               return 1500000LL;  // MEDIUM
    }
}

// ---- Color math ----

// RGB to HSV (H: 0-360, S: 0-1, V: 0-1)
static void rgb_to_hsv(led_color_t rgb, float *h, float *s, float *v)
{
    float r = rgb.r / 255.0f, g = rgb.g / 255.0f, b = rgb.b / 255.0f;
    float mx = r > g ? (r > b ? r : b) : (g > b ? g : b);
    float mn = r < g ? (r < b ? r : b) : (g < b ? g : b);
    float delta = mx - mn;
    *v = mx;
    *s = (mx > 0.001f) ? (delta / mx) : 0.0f;
    if (delta < 0.001f) {
        *h = 0;
    } else if (mx == r) {
        *h = 60.0f * fmodf((g - b) / delta, 6.0f);
    } else if (mx == g) {
        *h = 60.0f * ((b - r) / delta + 2.0f);
    } else {
        *h = 60.0f * ((r - g) / delta + 4.0f);
    }
    if (*h < 0) *h += 360.0f;
}

// HSV to RGB
static led_color_t hsv_to_rgb(float h, float s, float v)
{
    if (h < 0) h += 360.0f;
    h = fmodf(h, 360.0f);
    float c = v * s;
    float x = c * (1.0f - fabsf(fmodf(h / 60.0f, 2.0f) - 1.0f));
    float m = v - c;
    float r, g, b;
    if (h < 60)       { r = c; g = x; b = 0; }
    else if (h < 120) { r = x; g = c; b = 0; }
    else if (h < 180) { r = 0; g = c; b = x; }
    else if (h < 240) { r = 0; g = x; b = c; }
    else if (h < 300) { r = x; g = 0; b = c; }
    else              { r = c; g = 0; b = x; }
    return (led_color_t){
        (uint8_t)((r + m) * 255),
        (uint8_t)((g + m) * 255),
        (uint8_t)((b + m) * 255)
    };
}

// Scale color by brightness
static led_color_t scale_color(led_color_t c, uint8_t brightness)
{
    return (led_color_t){
        (uint8_t)((c.r * brightness) / 255),
        (uint8_t)((c.g * brightness) / 255),
        (uint8_t)((c.b * brightness) / 255)
    };
}

// ---- Animation: single LED ----

// Returns the color for one LED given phase [0..1) within one cycle
static led_color_t animate_single(led_color_t base, float phase,
                                  led_anim_t anim, led_style_t sty)
{
    float h, s, v;
    rgb_to_hsv(base, &h, &s, &v);

    if (sty == LED_STYLE_PURE) {
        // Color fixed, brightness varies
        switch (anim) {
            case LED_ANIM_BREATH:
                // Brightness 0 → max → 0
                v *= (0.5f + 0.5f * cosf(phase * 2.0f * (float)M_PI));
                break;
            case LED_ANIM_SCAN:
                // Brightness max → 0 → hold 0 → back to max
                if (phase < 0.7f) {
                    v *= (1.0f - phase / 0.7f);
                } else {
                    v = 0;
                }
                break;
            case LED_ANIM_GRADIENT:
                // Brightness wobble around max (±30%)
                v *= (0.7f + 0.3f * sinf(phase * 2.0f * (float)M_PI));
                break;
        }
        return scale_color(hsv_to_rgb(h, s, v), 1);
    } else {
        // COLORFUL: brightness fixed, hue varies
        switch (anim) {
            case LED_ANIM_BREATH:
                // Hue oscillate ±60°
                h += 60.0f * sinf(phase * 2.0f * (float)M_PI);
                break;
            case LED_ANIM_SCAN:
                // Hue rotate 360° then hold
                if (phase < 0.7f) {
                    h += (phase / 0.7f) * 360.0f;
                }
                // else hold at last hue
                break;
            case LED_ANIM_GRADIENT:
                // Hue wobble ±30°
                h += 30.0f * sinf(phase * 2.0f * (float)M_PI);
                break;
        }
        return scale_color(hsv_to_rgb(h, s, v), 1);
    }
}

// ---- Animation: outer ring (8 LEDs) ----

// Returns array of 8 colors for outer ring
static void animate_outer(led_color_t base, float phase,
                          led_anim_t anim, led_style_t sty,
                          led_color_t out[OUTER_COUNT])
{
    switch (anim) {
        case LED_ANIM_BREATH:
            // All 8 LEDs same animation, no phase offset
            for (int i = 0; i < OUTER_COUNT; i++) {
                out[i] = animate_single(base, phase, LED_ANIM_BREATH, sty);
            }
            break;
        case LED_ANIM_SCAN: {
            // Bright peak position rotates; trailing LEDs fade
            float peak_pos = phase * OUTER_COUNT;  // 0..8
            for (int i = 0; i < OUTER_COUNT; i++) {
                float dist = peak_pos - (float)i;
                if (dist < 0) dist += OUTER_COUNT;
                float fade = powf(0.5f, dist);
                if (sty == LED_STYLE_PURE) {
                    out[i] = scale_color(base, (uint8_t)(fade * 255));
                } else {
                    float h, s, v;
                    rgb_to_hsv(base, &h, &s, &v);
                    h += dist * 45.0f;  // Color shift along trail
                    out[i] = scale_color(hsv_to_rgb(h, s, v * fade), 1);
                }
            }
            break;
        }
        case LED_ANIM_GRADIENT:
            // Each LED has phase offset: i/8
            for (int i = 0; i < OUTER_COUNT; i++) {
                float offset_phase = fmodf(phase + (float)i / OUTER_COUNT, 1.0f);
                out[i] = animate_single(base, offset_phase, LED_ANIM_GRADIENT, sty);
            }
            break;
    }
}

// ---- Frame callback ----

static void frame_callback(void *arg)
{
    int64_t now = esp_timer_get_time();
    float phase = (float)((now - anim_start_us) % period_us()) / (float)period_us();

    rgb_t pixels[LED_COUNT] = {{0}};

    if (led_state == LED_STATE_PLAYING_OUTER) {
        // Center off, outer ring animated
        led_color_t outer[OUTER_COUNT];
        animate_outer(current_color, phase, animation, style, outer);
        uint8_t brt = brightness_value();
        for (int i = 0; i < OUTER_COUNT; i++) {
            pixels[i + 1] = scale_color(outer[i], brt);
        }

        // Check if one full scan rotation completed
        int64_t elapsed = now - anim_start_us;
        if (animation == LED_ANIM_SCAN) {
            // 1 round = outer_count steps at current speed
            int64_t one_round_us = period_us();
            if (elapsed >= one_round_us) {
                led_state = LED_STATE_IDLE;
                memset(pixels, 0, sizeof(pixels));
                esp_timer_stop(frame_timer);
            }
        } else {
            // Non-scan: play for 1 period then stop
            if (elapsed >= period_us()) {
                led_state = LED_STATE_IDLE;
                memset(pixels, 0, sizeof(pixels));
                esp_timer_stop(frame_timer);
            }
        }
    } else if (led_state == LED_STATE_WAITING_CENTER) {
        // Center animated, outer off
        led_color_t c = animate_single(current_color, phase, animation, style);
        pixels[CENTER_IDX] = scale_color(c, brightness_value());
    } else if (led_state == LED_STATE_DEMO) {
        // Both center and outer animated
        led_color_t c = animate_single(demo_color, phase, animation, style);
        pixels[CENTER_IDX] = scale_color(c, brightness_value());
        led_color_t outer[OUTER_COUNT];
        animate_outer(demo_color, phase, animation, style, outer);
        uint8_t brt = brightness_value();
        for (int i = 0; i < OUTER_COUNT; i++) {
            pixels[i + 1] = scale_color(outer[i], brt);
        }
    }

    ws2812_set_pixels(pixels, LED_COUNT);
}

static void start_timer(void)
{
    if (!frame_timer) {
        const esp_timer_create_args_t args = {
            .callback = &frame_callback,
            .name = "led_frame"
        };
        esp_timer_create(&args, &frame_timer);
    }
    anim_start_us = esp_timer_get_time();
    frame_count = 0;
    esp_timer_start_periodic(frame_timer, FRAME_INTERVAL_MS * 1000);
}

static void stop_timer(void)
{
    if (frame_timer) {
        esp_timer_stop(frame_timer);
    }
    // Clear all LEDs
    rgb_t pixels[LED_COUNT] = {{0}};
    ws2812_set_pixels(pixels, LED_COUNT);
}

// ---- Public API ----

void led_service_init(void)
{
    int32_t val;

    val = 1;
    storage_load_int(STORAGE_NAMESPACE_SETTINGS, "led_on", &val);
    enabled = (val != 0);

    val = 5;
    storage_load_int(STORAGE_NAMESPACE_SETTINGS, "led_bright", &val);
    brightness_level = (uint8_t)val;
    if (brightness_level > 10) brightness_level = 10;
    if (brightness_level < 1) brightness_level = 1;

    val = LED_SPEED_MEDIUM;
    storage_load_int(STORAGE_NAMESPACE_SETTINGS, "led_speed", &val);
    speed = (led_speed_t)val;

    val = LED_STYLE_PURE;
    storage_load_int(STORAGE_NAMESPACE_SETTINGS, "led_style", &val);
    style = (led_style_t)val;

    val = LED_ANIM_SCAN;
    storage_load_int(STORAGE_NAMESPACE_SETTINGS, "led_anim", &val);
    animation = (led_anim_t)val;

    ESP_LOGI(TAG, "LED service init: on=%d bright=%d speed=%d style=%d anim=%d",
             enabled, brightness_level, speed, style, animation);
}

void led_service_play(led_color_t color)
{
    if (!enabled) return;
    had_wait_before_play = (led_state == LED_STATE_WAITING_CENTER);
    if (had_wait_before_play) {
        saved_wait_color = current_color;
    }
    current_color = color;
    led_state = LED_STATE_PLAYING_OUTER;
    start_timer();
}

void led_service_wait(led_color_t color)
{
    if (!enabled) return;
    current_color = color;
    led_state = LED_STATE_WAITING_CENTER;
    start_timer();
}

void led_service_stop(void)
{
    led_state = LED_STATE_IDLE;
    stop_timer();
}

void led_service_set_enabled(bool on)
{
    enabled = on;
    storage_save_int(STORAGE_NAMESPACE_SETTINGS, "led_on", on ? 1 : 0);
    if (!on) {
        led_state = LED_STATE_IDLE;
        stop_timer();
    }
    ESP_LOGI(TAG, "LED %s", on ? "on" : "off");
}

bool led_service_is_enabled(void) { return enabled; }

void led_service_set_brightness(uint8_t level)
{
    if (level < 1) level = 1;
    if (level > 10) level = 10;
    brightness_level = level;
    storage_save_int(STORAGE_NAMESPACE_SETTINGS, "led_bright", level);
}

uint8_t led_service_get_brightness(void) { return brightness_level; }

void led_service_set_animation(led_anim_t anim)
{
    animation = anim;
    storage_save_int(STORAGE_NAMESPACE_SETTINGS, "led_anim", (int32_t)anim);
}

led_anim_t led_service_get_animation(void) { return animation; }

void led_service_set_style(led_style_t sty)
{
    style = sty;
    storage_save_int(STORAGE_NAMESPACE_SETTINGS, "led_style", (int32_t)sty);
}

led_style_t led_service_get_style(void) { return style; }

void led_service_set_speed(led_speed_t spd)
{
    speed = spd;
    storage_save_int(STORAGE_NAMESPACE_SETTINGS, "led_speed", (int32_t)spd);
}

led_speed_t led_service_get_speed(void) { return speed; }

void led_service_demo_start(led_color_t color)
{
    if (!enabled) return;
    // Save current state
    if (led_state == LED_STATE_WAITING_CENTER) {
        saved_wait_color = current_color;
        had_wait_before_play = true;
    } else {
        had_wait_before_play = false;
    }
    demo_color = color;
    led_state = LED_STATE_DEMO;
    start_timer();
}

void led_service_demo_change_color(led_color_t color)
{
    demo_color = color;
}

void led_service_demo_stop(void)
{
    led_state = LED_STATE_IDLE;
    stop_timer();
    // Restore previous wait if any
    if (had_wait_before_play) {
        led_service_wait(saved_wait_color);
    }
}

bool led_service_is_demo_active(void)
{
    return led_state == LED_STATE_DEMO;
}
```
